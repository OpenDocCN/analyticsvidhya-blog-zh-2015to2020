def isauthor(l):
    pattern = [
        '([\w]+):', 
        '([\w]+[\s]+[\w]+):',
        '([\w]+[\s]+[\w]+[\s]+[\w]+):',
        '([+]\d{2} \d{5} \d{5}):'
    ]

    result = re.match(patterns, l)
    if result:
        return True
    return False

    DateTime = DT.split(', ')
