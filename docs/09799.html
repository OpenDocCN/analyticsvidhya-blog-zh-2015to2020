def download_breed_images(breed_img):
    try:
        urllib.request.urlretrieve(breed_img[2], 
                                   os.path.join('rabbit_breeds/',                                                str(breed_img[1]),
                                                str(breed_img[1]) +
                                                str(breed_img[0]) +
                                                '.jpg'))
     except urllib.error.HTTPError as err:
         print(err.code)
