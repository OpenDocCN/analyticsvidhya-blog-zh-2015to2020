    pieces.add(new Piece(4, 7, true , Type.K));
    for (int i = 0; i < 2; i++) {
      pieces.add(new Piece(0 + i * 7, 0, false, Type.R));
      pieces.add(new Piece(0 + i * 7, 7, true , Type.R));

    }

    }
  }  Piece pieceAt(int c, int r) {

        return p;

      for (int c = 0; c < 8; c++) {
        Piece p = pieceAt(c, r);
        if (p == null) {
          brdStr += " .";

          switch (p.t) {

          case K: brdStr += p.w ? " K" : " k"; break;
          }
        }
      }
